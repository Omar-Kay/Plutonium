83 template<typename T>
84 class SigmoidIncrementer {
85 private:
86 double f;
87 double f_incr;
88 T target_initial_val;
89 T target_incr;
90 bool inverted_for_zero;
91
92 inline double ComputeIncrement() {
93 return (double)this->target_incr * (1.0f / (1.0f + exp(-this->f)));
94 }
95
96 public:
97 static constexpr double AllowedError = 0.4f;
98
99 constexpr SigmoidIncrementer() : f(0.0f), f_incr(0.0f), target_initial_val(0), target_incr(0) {}
100
101 void Start(const u32 f_steps, const T target_initial_val, const T target_incr) {
102 this->target_initial_val = target_initial_val;
103 this->target_incr = target_incr;
104 this->inverted_for_zero = false;
105
106 auto target_final_val = target_initial_val + target_incr;
107 if(target_final_val == 0) {
108 // Compute from 0 to X instead of from X to 0
109 this->inverted_for_zero = true;
110 this->target_initial_val = 0;
111 this->target_incr = target_initial_val;
112 target_final_val = target_initial_val;
113 }
114
115 const auto f_limit_abs = log((abs((double)target_final_val) - AllowedError) / AllowedError);
116 this->f_incr = (2.0f * f_limit_abs) / (double)f_steps;
117 this->f = -f_limit_abs;
118 }
119
120 inline void StartFromZero(const u32 f_steps, const T target_final_val) {
121 this->Start(f_steps, 0, target_final_val);
122 }
123
124 inline void StartToZero(const u32 f_steps, const T target_initial_val) {
125 this->Start(f_steps, target_initial_val, -target_initial_val);
126 }
127
128 bool Increment(T &target) {
129 if(this->IsDone()) {
130 return false;
131 }
132
133 const auto target_f = (double)this->target_initial_val + this->ComputeIncrement();
134 if(this->target_incr > 0) {
135 target = (T)((i32)(target_f + 0.5f));
136 }
137 else {
138 target = (T)((i32)(target_f - 0.5f));
139 }
140 this->f += this->f_incr;
141
142 const auto is_done = abs((double)target) >= abs((double)(this->target_initial_val + this->target_incr));
143 if(is_done) {
144 if(this->inverted_for_zero) {
145 target = 0;
146 }
147 else {
148 target = this->target_initial_val + this->target_incr;
149 }
150 this->target_incr = 0;
151 return true;
152 }
153 else if(this->inverted_for_zero) {
154 target = this->target_incr - target;
155 }
156
157 return false;
158 }
159
160 inline bool IsDone() {
161 return this->target_incr == 0;
162 }
163 };
