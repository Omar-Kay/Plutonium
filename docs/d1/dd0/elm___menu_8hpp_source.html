183 inline void RunSelectedItemCallback(const u64 keys) {
184 auto item = this->items.at(this->selected_item_idx);
185 const auto cb_count = item->GetOnKeyCallbackCount();
186 for(u32 i = 0; i < cb_count; i++) {
187 if(keys & item->GetOnKeyCallbackKey(i)) {
188 if(!this->cooldown_enabled) {
189 auto cb = item->GetOnKeyCallback(i);
190 if(cb) {
191 cb();
192 }
193 }
194 }
195 }
196 this->cooldown_enabled = false;
